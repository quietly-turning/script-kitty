## 1.  a simple query
SELECT * FROM institutions




## 2.  a simple query with a single condition
SELECT * FROM institutions WHERE city = "Athens"




## 3.  difference between "=" and "like" (and how to use "like")
#  3a. this returns nothing
SELECT * FROM institutions WHERE chief = "stan"

#  3b. this returns nothing
SELECT * FROM institutions WHERE chief LIKE "stan"

#  3c. this returns 25 rows
SELECT * FROM institutions WHERE chief LIKE "%stan%"






## 4.  a query with two conditions chained by "or"
# this returns 23 various rows
SELECT *
FROM institutions
WHERE institutions.chief LIKE "%Laura%"
OR institutions.chief LIKE "%Carrol%




## 5.  a query with two conditions chained by "and"
# this returns a single row where the chief is "Laura Carroll"
SELECT *
FROM institutions
WHERE institutions.chief LIKE "%Laura%"
AND institutions.chief LIKE "%Carrol%




## 6. a query that uses "does not equal" and has two conditions chained together by "and"
# this returns a single rural college in Houston, Missouri
SELECT *
FROM institutions
WHERE institutions.city = "Houston"
AND institutions.state != "TX"





## 7.  a simple query that only pulls specific columns into the results
# this returns the (name, city, state) of institutions where the chief's name contains "Victor"
SELECT name, city, state
FROM institutions
WHERE chief LIKE "%Victor%"




## 8.  another query that only pulls specific columns into the results
# this returns NY institutions with the word "medical" in the name
SELECT name, city, state
FROM institutions
WHERE name LIKE "%Medical%"
AND state = "NY"





## 9. complex query that pulls from multiple tables (and how to avoid cartesian products)
# this returns the nineteen Penn State campuses and their respective locales
SELECT institutions.name, institutions.state, locales.name
FROM institutions, locales
WHERE institutions.name LIKE "%Penn State%"
AND locales.id = institutions.locale_id





## 10. another complex query that pulls from multiple tables 
# this returns four unique institutions (coincidentally from four unique states) where the city is "Sterling" 
SELECT institutions.name, institutions.state, controls.name
FROM institutions, controls
WHERE institutions.city = "Sterling"
AND controls.id = institutions.control_id
